#include <msp430g2553.h>

int Deboucing(int porta)
{
  x = P1IN;
  x &= porta;
  for(i=0;i<5000;i++)
  {
    if((P1IN&porta) != x))
      {
        i = 0; //Houve variação na entrada, deve-se reiniciar a contagem;
      }
  }
  return x;
}
